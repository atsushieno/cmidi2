702static inline void cmidi2_ump_sysex_get_packet_of(uint64_t* result1, uint64_t* result2, uint8_t group, uint8_t numBytes, const void* srcData, int32_t index,
703 enum cmidi2_message_type messageType, int radix, bool hasStreamId, uint8_t streamId) {
704 uint8_t dst8[16];
705 memset(dst8, 0, 16);
706 const uint8_t *src8 = (const uint8_t*) srcData;
707
708 dst8[0] = (messageType << 4) + (group & 0xF);
709
710 enum cmidi2_sysex_status status;
711 uint8_t size;
712 if (numBytes <= radix) {
713 status = CMIDI2_SYSEX_IN_ONE_UMP;
714 size = numBytes; // single packet message
715 } else if (index == 0) {
716 status = CMIDI2_SYSEX_START;
717 size = radix;
718 } else {
719 uint8_t isEnd = index == cmidi2_ump_sysex_get_num_packets(numBytes, radix) - 1;
720 if (isEnd) {
721 size = numBytes % radix ? numBytes % radix : radix;
722 status = CMIDI2_SYSEX_END;
723 } else {
724 size = radix;
725 status = CMIDI2_SYSEX_CONTINUE;
726 }
727 }
728 dst8[1] = status + size + (hasStreamId ? 1 : 0);
729
730 if (hasStreamId)
731 dst8[2] = streamId;
732
733 uint8_t dstOffset = hasStreamId ? 3 : 2;
734 for (uint8_t i = 0, j = index * radix; i < size; i++, j++)
735 dst8[i + dstOffset] = src8[j];
736
737 *result1 = cmidi2_ump_read_uint64_bytes_be(dst8);
738 if (result2)
739 *result2 = cmidi2_ump_read_uint64_bytes_be(dst8 + 8);
740}

842 dst8[0] = (CMIDI2_MESSAGE_TYPE_SYSEX8_MDS << 4) + (group & 0xF);
843 dst8[1] = CMIDI2_MIXED_DATA_STATUS_HEADER + mdsId;
844 *((uint16_t*) (void*) (dst8 + 2)) = numBytesInChunk;
845 *((uint16_t*) (void*) (dst8 + 4)) = numChunks;
846 *((uint16_t*) (void*) (dst8 + 6)) = chunkIndex;
847 *((uint16_t*) (void*) (dst8 + 8)) = manufacturerId;
848 *((uint16_t*) (void*) (dst8 + 10)) = deviceId;
849 *((uint16_t*) (void*) (dst8 + 12)) = subId;
850 *((uint16_t*) (void*) (dst8 + 14)) = subId2;
851
852 *result1 = cmidi2_ump_read_uint64_bytes_be(dst8);
853 if (result2)
854 *result2 = cmidi2_ump_read_uint64_bytes_be(dst8 + 8);
855}

857// srcData points to exact start of the source data.
858static inline void cmidi2_ump_mds_get_payload_of(uint8_t group, uint8_t mdsId, uint16_t numBytes, const void* srcData, uint64_t* result1, uint64_t* result2) {
859 uint8_t dst8[16];
860 memset(dst8, 0, 16);
861 const uint8_t *src8 = (const uint8_t*) srcData;
862
863 dst8[0] = (CMIDI2_MESSAGE_TYPE_SYSEX8_MDS << 4) + (group & 0xF);
864 dst8[1] = CMIDI2_MIXED_DATA_STATUS_PAYLOAD + mdsId;
865
866 uint8_t radix = 14;
867 uint8_t size = numBytes < radix ? numBytes % radix : radix;
868
869 for (uint8_t i = 0; i < size; i++)
870 dst8[i + 2] = src8[i];
871
872 *result1 = cmidi2_ump_read_uint64_bytes_be(dst8);
873 if (result2)
874 *result2 = cmidi2_ump_read_uint64_bytes_be(dst8 + 8);
875}

877/* process() - more complicated function */
878// This returns NULL for success, or anything else for failure.
879typedef void*(*cmidi2_mds_handler)(uint64_t data1, uint64_t data2, size_t chunkId, size_t payloadId, void* context);
880
881// This returns NULL for success, or anything else that `sendUMP` returns for failure.
882static inline void* cmidi2_ump_mds_process(uint8_t group, uint8_t mdsId, void* data, uint32_t length, cmidi2_mds_handler sendUMP, void* context)
883{
884 int32_t numChunks = cmidi2_ump_mds_get_num_chunks(length);
885 for (int c = 0; c < numChunks; c++) {
886 int32_t maxChunkSize = 14 * 65535;
887 int32_t chunkSize = c + 1 == numChunks ? (int32_t)(length % maxChunkSize) : maxChunkSize;
888 int32_t numPayloads = cmidi2_ump_mds_get_num_payloads(chunkSize);
889 for (int p = 0; p < numPayloads; p++) {
890 uint64_t result1, result2;
891 size_t offset = 14 * (65536 * c + p);
892 cmidi2_ump_mds_get_payload_of(group, mdsId, chunkSize, (uint8_t*) data + offset, &result1, &result2);
893 void* retCode = sendUMP(result1, result2, c, p, context);
894 if (retCode)
895 return retCode;
896 }
897 }
898 return NULL;
899}

903static inline uint16_t cmidi2_ump_flex_data_get_num_packets(uint32_t numTotalBytesInFlexData) {
904 return numTotalBytesInFlexData / 12 + (numTotalBytesInFlexData % 12 ? 1 : 0);

907static inline void cmidi2_ump_flex_data_complete_packet(uint8_t group, uint8_t addressing, uint8_t channel,
908 uint8_t statusBank, uint8_t statusCode, uint32_t data1, uint32_t data2, uint32_t data3,
909 uint64_t* result1, uint64_t* result2) {
910 uint8_t dst8[4];
911 memset(dst8, 0, 4);
912 dst8[0] = (CMIDI2_MESSAGE_TYPE_FLEX_DATA << 4) + (group & 0xF);
913 dst8[1] = (CMIDI2_SYSEX_IN_ONE_UMP << 2) + ((addressing & 0x3) << 4) + (channel & 0xF);
914 dst8[2] = statusBank;
915 dst8[3] = statusCode;
916
917 *result1 = ((uint64_t) cmidi2_ump_read_uint32_bytes_be(dst8) << 32) + data1;
918 *result2 = ((uint64_t) data2 << 32) + data3;
919}

921static inline void cmidi2_ump_flex_data_get_packet_of(uint8_t group, uint8_t addressing, uint8_t channel,
922 uint8_t statusBank, uint8_t statusCode, uint16_t numBytes, const void* srcData, int32_t currentPacket,
923 uint64_t* result1, uint64_t* result2) {
924 uint8_t dst8[16];
925 memset(dst8, 0, 16);
926 const uint8_t *src8 = (const uint8_t*) srcData;
927
928 dst8[0] = (CMIDI2_MESSAGE_TYPE_FLEX_DATA << 4) + (group & 0xF);
929
930 const size_t radix = 12;
931 enum cmidi2_sysex_status format;
932 uint8_t size;
933 if (numBytes <= radix) {
934 format = CMIDI2_SYSEX_IN_ONE_UMP;
935 size = numBytes; // single packet message
936 } else if (currentPacket == 0) {
937 format = CMIDI2_SYSEX_START;
938 size = radix;
939 } else {
940 uint8_t isEnd = currentPacket == cmidi2_ump_sysex_get_num_packets(numBytes, radix) - 1;
941 if (isEnd) {
942 size = numBytes % radix ? numBytes % radix : radix;
943 format = CMIDI2_SYSEX_END;
944 } else {
945 size = radix;
946 format = CMIDI2_SYSEX_CONTINUE;
947 }
948 }
949 dst8[1] = (format << 2) + ((addressing & 0x3) << 4) + (channel & 0xF);
950 dst8[2] = statusBank;
951 dst8[3] = statusCode;
952
953 for (uint8_t i = 0; i < size; i++)
954 dst8[i + 4] = src8[i];
955
956 *result1 = cmidi2_ump_read_uint64_bytes_be(dst8);
957 *result2 = cmidi2_ump_read_uint64_bytes_be(dst8 + 8);
958}

960/* process() - more complicated function */
961// This returns NULL for success, or anything else for failure.
962typedef void* (*cmidi2_flex_data_handler)(uint64_t data1, uint64_t data2, void* context);
963
964// This returns NULL for success, or anything else that `sendUMP` returns for failure.
965// `text` is usually a null-terminated text string, but it may contain `\0` as Melisma in lyricText. Hence we still need `length`.
966static inline void* cmidi2_ump_flex_data_process(uint8_t group, uint8_t addressing, uint8_t channel,
967 uint8_t statusBank, uint8_t statusCode, const char* text, uint32_t length, cmidi2_flex_data_handler sendUMP, void* context)
968{
969 int32_t numPackets = cmidi2_ump_flex_data_get_num_packets(length);
970 for (int p = 0; p < numPackets; p++) {
971 uint64_t result1, result2;
972 cmidi2_ump_flex_data_get_packet_of(group, addressing, channel, statusBank, statusCode, length, text, p, &result1, &result2);
973 void *retCode = sendUMP(result1, result2, context);
974 if (retCode != 0)
975 return retCode;
976 }
977 return NULL;
978}

1302// --------
1303// Realtime-safe SysEx8 binary reader
1304//
1305// The primary end-user facing function is `cmidi2_ump_get_sysex8_data()`, but the
1306// function signature needs some explanation as it requires some complicated processing.
1307//
1308// There can be multiple SysEx8 "streams" that simultaneously run within a UMP stream
1309// (e.g. sysex8 start packet for stream #1, sysex8 continue packet for stream #1,
1310// sysex8 start packet for stream #2, sysex8 continue packet for stream #1,
1311// sysex8 continue packet for stream #2, sysex8 complete packet for stream #3,
1312// sysex8 end packet for stream #2, sysex8 end packet for strean #1 ...).
1313//
1314// Those stream state must be preserved during UMP parsing. It is represented as
1315// `cmidi2_ump_binary_read_state` struct.
1316//
1317// There are handful of usage scenarios where we...
1318// - want to handle them all
1319// - want to handle only relevant ones
1320// - can control what we generate and determine that only one stream can appear so
1321// the context stream is always obvious
1322//
1323// To handle them all, we need "stream selector". There is a function type
1324// `cmidi2_ump_stream_selector_func`, and a function pointer is passed to the entry
1325// point `cmidi2_ump_get_sysex8_data()` function.
1326//
1327// Another control point is where we want to finish parsing sysex8. Whenever a relevant
1328// stream is parsed, `cmidi2_ump_get_syex8_data()` calls "continuity checker" function
1329// which is typed as `cmidi2_ump_binary_read_continuity_checker_func`.
1330// If this function returns true, then it completes parsing, returning the number of
1331// the parsed UMPs (in uint32_t length).
1332//
1333// See `testType5Messages_sysex8_reader_writer()` testcase for the actual usage example.
1334//
1335
1336enum cmidi2_ump_binary_reader_result_code {
1337 CMIDI2_BINARY_READER_RESULT_INCOMPLETE = 0,
1338 CMIDI2_BINARY_READER_RESULT_COMPLETE = 1,
1339 CMIDI2_BINARY_READER_RESULT_NO_SPACE = 2,
1340};

1385// it is a special copy function that only works with sysex8 memory state at `src` that can access beyond `sizeInBytes`.
1386static inline void cmidi2_internal_sysex8_copy_data_byte_swapping(uint8_t* dst, uint8_t srcHead, uint32_t* srcTail, size_t sizeInBytes) {
1387 if (sizeInBytes == 0)
1388 return;
1389
1390 // the first byte is at the lowest byte so it is always safe to copy as is.
1391 dst[0] = srcHead;
1392 uint8_t* d = dst + 1;
1393 uint32_t* s = srcTail;
1394 sizeInBytes--;
1395
1396 // copy the rest
1397 size_t i = 0;
1398 while (i + 3 < sizeInBytes) {
1399 uint32_t i32 = *s;
1400 d[i++] = i32 >> 24;
1401 d[i++] = (i32 >> 16) & 0xFF;
1402 d[i++] = (i32 >> 8) & 0xFF;
1403 d[i++] = i32 & 0xFF;
1404 s++;
1405 }
1406 for (; i < sizeInBytes; i++)
1407 d[i] = s[i + 3 - i % 4];
1408}

1412static inline size_t cmidi2_ump_get_sysex8_data(
1413 cmidi2_ump_stream_selector_func streamSelector,
1414 void* streamSelectorContext,
1415 cmidi2_ump_binary_read_continuity_checker_func continuityChecker,
1416 const cmidi2_ump* ump,
1417 const size_t umpCapacityInInt) {
1418
1419 cmidi2_ump *umpPtr = (cmidi2_ump*) ump, *umpEnd = (cmidi2_ump*) ump + umpCapacityInInt;
1420 for (;umpPtr < umpEnd; umpPtr += cmidi2_ump_get_message_size_bytes(umpPtr) / sizeof(cmidi2_ump)) {
1421 if (cmidi2_ump_get_message_type(umpPtr) != CMIDI2_MESSAGE_TYPE_SYSEX8_MDS)
1422 continue;
1423 switch (cmidi2_ump_get_status_code(umpPtr)) {
1424 case CMIDI2_SYSEX_IN_ONE_UMP:
1425 case CMIDI2_SYSEX_START:
1426 case CMIDI2_SYSEX_END:
1427 case CMIDI2_SYSEX_CONTINUE:
1428 break;
1429 default:
1430 continue; // the only expected value here is MDS (8 or 9)
1431 }
1432
1433 cmidi2_ump_binary_read_state *state = streamSelector(cmidi2_ump_get_sysex8_stream_id(umpPtr), streamSelectorContext);
1434 if (state == NULL)
1435 continue;
1436
1437 size_t copySize = cmidi2_ump_get_sysex8_num_bytes(umpPtr) - 1; // SysEx8 size field contains the size byte itself, hence -1.
1438 if (state->dataSize + copySize >= state->dataCapacity) {
1439 state->resultCode = CMIDI2_BINARY_READER_RESULT_NO_SPACE;
1440 return umpPtr - ump;
1441 }
1442
1443 if (cmidi2_util_is_platform_little_endian())
1444 // We need to swap data along with byte order for resulting data...
1445 cmidi2_internal_sysex8_copy_data_byte_swapping(state->data + state->dataSize, (*umpPtr) & 0xFF, umpPtr + 1, copySize);
1446 else
1447 memcpy(state->data + state->dataSize, ((uint8_t*) (void*) umpPtr) + 3, copySize);
1448
1449 state->dataSize += copySize;
1450
1451 if (continuityChecker != NULL)
1452 if (continuityChecker(state, umpPtr))
1453 return umpPtr + cmidi2_ump_get_message_size_bytes(umpPtr) / sizeof(cmidi2_ump) - ump; // "break here" is indicated
1454
1455 // otherwise default continuity checker
1456 switch (cmidi2_ump_get_status_code(umpPtr)) {
1457 case CMIDI2_SYSEX_IN_ONE_UMP:
1458 case CMIDI2_SYSEX_END:
1459 state->resultCode = CMIDI2_BINARY_READER_RESULT_COMPLETE;
1460 if (state->continueOnCompletion)
1461 return umpPtr + cmidi2_ump_get_message_size_bytes(umpPtr) / sizeof(cmidi2_ump) - ump; // default "break here" condition.
1462 break;
1463 }
1464 }
1465 // finished parsing while no stream indicated "break here" for completion.
1466 return umpPtr - ump;
1467}

2003static inline uint32_t cmidi2_midi1_get_7bit_encoded_int(uint8_t* bytes, uint32_t length)
2004{
2005 uint8_t* start = bytes;
2006 uint32_t value = 0;
2007 for (int digits = 0;; digits++) {
2008 if (bytes >= start + length)
2009 break;
2010 value += (0x7F & (*bytes)) << (digits * 7);
2011 if (*bytes < 0x80)
2012 break;
2013 bytes++;
2014 }
2015 bytes++;
2016 return value;
2017}

2019static inline uint32_t cmidi2_midi1_get_message_size(uint8_t* bytes, uint32_t length) {
2020 uint32_t metaLength;
2021 uint8_t* start = bytes;
2022 uint8_t* end = bytes + length;
2023 switch (bytes[0]) {
2024 case 0xF0:
2025 for (bytes++; bytes < end; bytes++)
2026 if (*bytes == 0xF7)
2027 break;
2028 bytes++;
2029 break;
2030 case 0xFF:
2031 bytes++;
2032 metaLength = cmidi2_midi1_get_7bit_encoded_int(bytes, end - bytes);
2033 bytes += metaLength + cmidi2_midi1_get_7bit_encoded_int_length(metaLength);
2034 break;
2035 default:
2036 switch (bytes[0] & 0xF0) {
2037 case 0xC0:
2038 case 0xD0:
2039 bytes += 2;
2040 break;
2041 default:
2042 bytes += 3;
2043 break;
2044 }
2045 break;
2046 }
2047 return bytes - start;

2753 cmidi2_ump_jr_timestamp_direct(deltaTime > 0xFFFF ? 0xFFFF : deltaTime));

2781 break;

2783 }
2784 if (seq1Idx >= (int64_t)seq1Size || seq2Idx >= (int64_t)seq2Size)
2785 break;

2787 timestamp1, timestamp2, &lastTimestamp, dst, &dIdx, dstCapacity))
2788 break;
2789
2790 if (timestamp1 <= timestamp2) {
