457 static inline void cmidi2_ump_mds_get_payload_of(uint8_t group, uint8_t mdsId, uint16_t numBytes, void* srcData, uint64_t* result1, uint64_t* result2) {
458  uint8_t dst8[16];
459  memset(dst8, 0, 16);
460  uint8_t *src8 = (uint8_t*) srcData;
461 
462  dst8[0] = (CMIDI2_MESSAGE_TYPE_SYSEX8_MDS << 4) + (group & 0xF);
463  dst8[1] = CMIDI2_MIXED_DATA_STATUS_PAYLOAD + mdsId;
464 
465  uint8_t radix = 14;
466  uint8_t size = numBytes < radix ? numBytes % radix : radix;
467 
468  for (uint8_t i = 0, j = 0; i < size; i++, j++)
469  dst8[i + 2] = src8[j];
470 
471  *result1 = cmidi2_ump_read_uint64_bytes(dst8);
472  if (result2)
473  *result2 = cmidi2_ump_read_uint64_bytes(dst8 + 8);
474 }
