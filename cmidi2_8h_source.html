430 static inline void cmidi2_ump_sysex_get_packet_of(uint64_t* result1, uint64_t* result2, uint8_t group, uint8_t numBytes, void* srcData, int32_t index,
431  enum cmidi2_message_type messageType, int radix, bool hasStreamId, uint8_t streamId) {
432  uint8_t dst8[16];
433  memset(dst8, 0, 16);
434  uint8_t *src8 = (uint8_t*) srcData;
435 
436  dst8[0] = (messageType << 4) + (group & 0xF);
437 
438  enum cmidi2_sysex_status status;
439  uint8_t size;
440  if (numBytes <= radix) {
441  status = CMIDI2_SYSEX_IN_ONE_UMP;
442  size = numBytes; // single packet message
443  } else if (index == 0) {
444  status = CMIDI2_SYSEX_START;
445  size = radix;
446  } else {
447  uint8_t isEnd = index == cmidi2_ump_sysex_get_num_packets(numBytes, radix) - 1;
448  if (isEnd) {
449  size = numBytes % radix ? numBytes % radix : radix;
450  status = CMIDI2_SYSEX_END;
451  } else {
452  size = radix;
453  status = CMIDI2_SYSEX_CONTINUE;
454  }
455  }
456  dst8[1] = status + size + (hasStreamId ? 1 : 0);
457 
458  if (hasStreamId)
459  dst8[2] = streamId;
460 
461  uint8_t dstOffset = hasStreamId ? 3 : 2;
462  for (uint8_t i = 0, j = index * radix; i < size; i++, j++)
463  dst8[i + dstOffset] = src8[j];
464 
465  *result1 = cmidi2_ump_read_uint64_bytes(dst8);
466  if (result2)
467  *result2 = cmidi2_ump_read_uint64_bytes(dst8 + 8);
468 }
