709 static inline void cmidi2_ump_sysex_get_packet_of(uint64_t* result1, uint64_t* result2, uint8_t group, uint8_t numBytes, const void* srcData, int32_t index,
710  enum cmidi2_message_type messageType, int radix, bool hasStreamId, uint8_t streamId) {
711  uint8_t dst8[16];
712  memset(dst8, 0, 16);
713  const uint8_t *src8 = (const uint8_t*) srcData;
714 
715  dst8[0] = (messageType << 4) + (group & 0xF);
716 
717  enum cmidi2_sysex_status status;
718  uint8_t size;
719  if (numBytes <= radix) {
720  status = CMIDI2_SYSEX_IN_ONE_UMP;
721  size = numBytes; // single packet message
722  } else if (index == 0) {
723  status = CMIDI2_SYSEX_START;
724  size = radix;
725  } else {
726  uint8_t isEnd = index == cmidi2_ump_sysex_get_num_packets(numBytes, radix) - 1;
727  if (isEnd) {
728  size = numBytes % radix ? numBytes % radix : radix;
729  status = CMIDI2_SYSEX_END;
730  } else {
731  size = radix;
732  status = CMIDI2_SYSEX_CONTINUE;
733  }
734  }
735  dst8[1] = status + size + (hasStreamId ? 1 : 0);
736 
737  if (hasStreamId)
738  dst8[2] = streamId;
739 
740  uint8_t dstOffset = hasStreamId ? 3 : 2;
741  for (uint8_t i = 0, j = index * radix; i < size; i++, j++)
742  dst8[i + dstOffset] = src8[j];
743 
744  *result1 = cmidi2_ump_read_uint64_bytes_be(dst8);
745  if (result2)
746  *result2 = cmidi2_ump_read_uint64_bytes_be(dst8 + 8);
747 }

909 static inline void cmidi2_ump_flex_data_complete_packet(uint8_t group, uint8_t addressing, uint8_t channel,
910  uint8_t statusBank, uint8_t statusCode, uint32_t data1, uint32_t data2, uint32_t data3,
911  uint64_t* result1, uint64_t* result2) {
912  uint8_t dst8[4];
913  memset(dst8, 0, 4);
914  dst8[0] = (CMIDI2_MESSAGE_TYPE_FLEX_DATA << 4) + (group & 0xF);
915  dst8[1] = (CMIDI2_SYSEX_IN_ONE_UMP << 2) + ((addressing & 0x3) << 4) + (channel & 0xF);
916  dst8[2] = statusBank;
917  dst8[3] = statusCode;
918

923 static inline void cmidi2_ump_flex_data_get_packet_of(uint8_t group, uint8_t addressing, uint8_t channel,
924  uint8_t statusBank, uint8_t statusCode, uint16_t numBytes, const void* srcData, int32_t currentPacket,
925  uint64_t* result1, uint64_t* result2) {
926  uint8_t dst8[16];
927  memset(dst8, 0, 16);
928  const uint8_t *src8 = (const uint8_t*) srcData;
929 
930  dst8[0] = (CMIDI2_MESSAGE_TYPE_FLEX_DATA << 4) + (group & 0xF);
931 
932  const size_t radix = 12;
933  enum cmidi2_sysex_status format;
934  uint8_t size;
935  if (numBytes <= radix) {
936  format = CMIDI2_SYSEX_IN_ONE_UMP;
937  size = numBytes; // single packet message
938  } else if (currentPacket == 0) {
939  format = CMIDI2_SYSEX_START;
940  size = radix;
941  } else {
942  uint8_t isEnd = currentPacket == cmidi2_ump_sysex_get_num_packets(numBytes, radix) - 1;
943  if (isEnd) {
944  size = numBytes % radix ? numBytes % radix : radix;
945  format = CMIDI2_SYSEX_END;
946  } else {
947  size = radix;
948  format = CMIDI2_SYSEX_CONTINUE;
949  }
950  }
951  dst8[1] = (format << 2) + ((addressing & 0x3) << 4) + (channel & 0xF);
952  dst8[2] = statusBank;
953  dst8[3] = statusCode;
954 
955  for (uint8_t i = 0; i < size; i++)
956  dst8[i + 4] = src8[i];
957 
958  *result1 = cmidi2_ump_read_uint64_bytes_be(dst8);
959  *result2 = cmidi2_ump_read_uint64_bytes_be(dst8 + 8);
960 }

1301 // Realtime-safe SysEx8 binary reader
1302 //
1303 // The primary end-user facing function is `cmidi2_ump_get_sysex8_data()`, but the
1304 // function signature needs some explanation as it requires some complicated processing.
1305 //
1306 // There can be multiple SysEx8 "streams" that simultaneously run within a UMP stream
1307 // (e.g. sysex8 start packet for stream #1, sysex8 continue packet for stream #1,
1308 // sysex8 start packet for stream #2, sysex8 continue packet for stream #1,
1309 // sysex8 continue packet for stream #2, sysex8 complete packet for stream #3,
1310 // sysex8 end packet for stream #2, sysex8 end packet for strean #1 ...).
1311 //
1312 // Those stream state must be preserved during UMP parsing. It is represented as
1313 // `cmidi2_ump_binary_read_state` struct.
1314 //
1315 // There are handful of usage scenarios where we...
1316 // - want to handle them all
1317 // - want to handle only relevant ones
1318 // - can control what we generate and determine that only one stream can appear so
1319 // the context stream is always obvious
1320 //
1321 // To handle them all, we need "stream selector". There is a function type
1322 // `cmidi2_ump_stream_selector_func`, and a function pointer is passed to the entry
1323 // point `cmidi2_ump_get_sysex8_data()` function.
1324 //
1325 // Another control point is where we want to finish parsing sysex8. Whenever a relevant
1326 // stream is parsed, `cmidi2_ump_get_syex8_data()` calls "continuity checker" function
1327 // which is typed as `cmidi2_ump_binary_read_continuity_checker_func`.
1328 // If this function returns true, then it completes parsing, returning the number of
1329 // the parsed UMPs (in uint32_t length).
1330 //
1331 // See `testType5Messages_sysex8_reader_writer()` testcase for the actual usage example.
1332 //

1383 // it is a special copy function that only works with sysex8 memory state at `src` that can access beyond `sizeInBytes`.
1384 static inline void cmidi2_internal_sysex8_copy_data_byte_swapping(uint8_t* dst, uint8_t srcHead, uint32_t* srcTail, size_t sizeInBytes) {
1385  if (sizeInBytes == 0)
1386  return;
1387 
1388  // the first byte is at the lowest byte so it is always safe to copy as is.
1389  dst[0] = srcHead;
1390  uint8_t* d = dst + 1;
1391  uint32_t* s = srcTail;
1392  sizeInBytes--;
1393 
1394  // copy the rest
1395  size_t i = 0;
1396  while (i + 3 < sizeInBytes) {
1397  uint32_t i32 = *s;
1398  d[i++] = i32 >> 24;
1399  d[i++] = (i32 >> 16) & 0xFF;
1400  d[i++] = (i32 >> 8) & 0xFF;
1401  d[i++] = i32 & 0xFF;
1402  s++;
1403  }
1404  for (; i < sizeInBytes; i++)
1405  d[i] = s[i + 3 - i % 4];
1406 }

1410 static inline size_t cmidi2_ump_get_sysex8_data(
1411  cmidi2_ump_stream_selector_func streamSelector,
1412  void* streamSelectorContext,
1413  cmidi2_ump_binary_read_continuity_checker_func continuityChecker,
1414  const cmidi2_ump* ump,
1415  const size_t umpCapacityInInt) {
1416 
1417  cmidi2_ump *umpPtr = (cmidi2_ump*) ump, *umpEnd = (cmidi2_ump*) ump + umpCapacityInInt;
1418  for (;umpPtr < umpEnd; umpPtr += cmidi2_ump_get_message_size_bytes(umpPtr) / sizeof(cmidi2_ump)) {
1419  if (cmidi2_ump_get_message_type(umpPtr) != CMIDI2_MESSAGE_TYPE_SYSEX8_MDS)
1420  continue;
1421  switch (cmidi2_ump_get_status_code(umpPtr)) {
1422  case CMIDI2_SYSEX_IN_ONE_UMP:
1423  case CMIDI2_SYSEX_START:
1424  case CMIDI2_SYSEX_END:
1425  case CMIDI2_SYSEX_CONTINUE:
1426  break;
1427  default:
1428  continue; // the only expected value here is MDS (8 or 9)
1429  }
1430 
1431  cmidi2_ump_binary_read_state *state = streamSelector(cmidi2_ump_get_sysex8_stream_id(umpPtr), streamSelectorContext);
1432  if (state == NULL)
1433  continue;
1434 
1435  size_t copySize = cmidi2_ump_get_sysex8_num_bytes(umpPtr) - 1; // SysEx8 size field contains the size byte itself, hence -1.
1436  if (state->dataSize + copySize >= state->dataCapacity) {
1437  state->resultCode = CMIDI2_BINARY_READER_RESULT_NO_SPACE;
1438  return umpPtr - ump;
1439  }
1440 
1441  if (cmidi2_util_is_platform_little_endian())
1442  // We need to swap data along with byte order for resulting data...
1443  cmidi2_internal_sysex8_copy_data_byte_swapping(state->data + state->dataSize, (*umpPtr) & 0xFF, umpPtr + 1, copySize);
1444  else
1445  memcpy(state->data + state->dataSize, ((uint8_t*) (void*) umpPtr) + 3, copySize);
1446 
1447  state->dataSize += copySize;
1448 
1449  if (continuityChecker != NULL)
1450  if (continuityChecker(state, umpPtr))
1451  return umpPtr + cmidi2_ump_get_message_size_bytes(umpPtr) / sizeof(cmidi2_ump) - ump; // "break here" is indicated
1452 
1453  // otherwise default continuity checker
1454  switch (cmidi2_ump_get_status_code(umpPtr)) {
1455  case CMIDI2_SYSEX_IN_ONE_UMP:
1456  case CMIDI2_SYSEX_END:
1457  state->resultCode = CMIDI2_BINARY_READER_RESULT_COMPLETE;
1458  if (state->continueOnCompletion)
1459  return umpPtr + cmidi2_ump_get_message_size_bytes(umpPtr) / sizeof(cmidi2_ump) - ump; // default "break here" condition.
1460  break;
1461  }
1462  }
1463  // finished parsing while no stream indicated "break here" for completion.
1464  return umpPtr - ump;
1465 }

2002 {
2003  uint8_t* start = bytes;
2004  uint32_t value = 0;
2005  for (int digits = 0;; digits++) {
2006  if (bytes >= start + length)
2007  break;
2008  value += (0x7F & (*bytes)) << (digits * 7);
2009  if (*bytes < 0x80)
2010  break;
2011  bytes++;
2012  }
2013  bytes++;
2014  return value;
2015 }

2018  uint32_t metaLength;
2019  uint8_t* start = bytes;
2020  uint8_t* end = bytes + length;
2021  switch (bytes[0]) {
2022  case 0xF0:
2023  for (bytes++; bytes < end; bytes++)
2024  if (*bytes == 0xF7)
2025  break;
2026  bytes++;
2027  break;
2028  case 0xFF:
2029  bytes++;
2030  metaLength = cmidi2_midi1_get_7bit_encoded_int(bytes, end - bytes);
2031  bytes += metaLength + cmidi2_midi1_get_7bit_encoded_int_length(metaLength);
2032  break;
2033  default:
2034  switch (bytes[0] & 0xF0) {
2035  case 0xC0:
2036  case 0xD0:
2037  bytes += 2;
2038  break;
2039  default:
2040  bytes += 3;
2041  break;
2042  }
2043  break;
2044  }
